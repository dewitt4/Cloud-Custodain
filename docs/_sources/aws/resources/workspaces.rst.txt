.. _aws.workspaces.filters.connection-status:

connection-status
+++++++++++++++++
Filter Workspaces based on user connection information

:example:

.. code-block:: yaml

        policies:

          - name: workspaces-abandoned
            resource: workspaces
            filters:
              - type: connection-status
                value_type: age
                key: LastKnownUserConnectionTimestamp
                op: ge
                value: 90

          - name: workspaces-expensive-zombies
            resource: workspaces
            filters:
              - "WorkspaceProperties.RunningMode": ALWAYS_ON
              - type: connection-status
                value_type: age
                key: LastKnownUserConnectionTimestamp
                op: ge
                value: 30
